import { signal, computed } from 'what-framework';
import { h, mount } from 'what-framework';

function Counter() {
  const count = signal(0);
  const doubled = computed(() => count() * 2);

  return h('div', {},
    h('button', {
      onClick: () => count.set(c => c + 1)
    }, 'Count: ', () => count()),
    h('p', {}, 'Doubled: ', () => doubled())
  );
}

mount(h(Counter), '#app');

Reactive Data Fetching
import { useSWR } from 'what-framework';

function UserProfile({ id }) {
  const { data, error, isLoading } = useSWR(
    `user-${id}`,
    (key, { signal }) =>
      fetch(`/api/users/${id}`, { signal })
        .then(r => r.json()),
    { revalidateOnFocus: true }
  );

  return h('div', {},
    () => isLoading()
      ? 'Loading...'
      : data()?.name
  );
}
Form Validation
import { useForm, rules } from 'what-framework';

function SignUp() {
  const { register, handleSubmit, formState }
    = useForm({ mode: 'onBlur' });

  return h('form', {
    onSubmit: handleSubmit(data => {
      console.log(data);
    })
  },
    h('input', register('email', {
      required: true,
      validate: rules.email()
    })),
    h('button', {}, 'Submit')
  );
}
Global Store
import { createStore, derived }
  from 'what-framework';

const useCart = createStore({
  items: [],
  total: derived(state =>
    state.items.reduce(
      (sum, i) => sum + i.price, 0
    )
  ),
  addItem(item) {
    this.items = [...this.items, item];
  },
  clear() {
    this.items = [];
  }
});
