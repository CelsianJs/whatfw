function Counter() {

    <div>

    </div>
  )
}

Why What

Everything you need.
Nothing you don't.

A complete framework that respects your bundle size. Zero dependencies. Fine-grained reactivity that only updates what changed.

Fine-Grained Signals

Reactive primitives that track exact dependencies. When a signal changes, only the DOM nodes that read it update.

const count = signal(0)
const doubled = computed(() => count() * 2)
// Only this text node re-renders:
<p>{() => doubled()}</p>

Islands Architecture

Ship zero JavaScript by default. Hydrate components on idle, visible, or first interaction. Perfect for content sites.

<Header />          // Static HTML
<Comments client:visible />
<Analytics client:idle />
<Menu client:interaction />

Examples

Code that reads clearly.

No magic. Just reactive values and clean APIs.

import { signal, computed, mount } from 'what-framework'

function Counter() {
  const count = signal(0)
  const doubled = computed(() => count() * 2)

  return (
    <div>
      <h1>{() => count()}</h1>
      <p>Doubled: {() => doubled()}</p>
      <button onclick={() => count(c => c + 1)}>Increment</button>
    </div>
  )
}

mount(<Counter />, '#app')
import { useSWR } from 'what-framework'

function UserProfile({ id }) {
  const { data, error, isLoading } = useSWR(
    `user-${id}`,
    () => fetch(`/api/users/${id}`).then(r => r.json()),
    { revalidateOnFocus: true }
  )

  return (
    <div>
      {() => isLoading()
        ? <p>Loading...</p>
        : <h2>{data()?.name}</h2>
      }
    </div>
  )
}
import { createStore, derived } from 'what-framework'

const useCart = createStore({
  items: [],

  total: derived(state =>
    state.items.reduce((sum, i) => sum + i.price, 0)
  ),

  addItem(item) {
    this.items = [...this.items, item]
  },

  clear() {
    this.items = []
  }
})

// Use anywhere — no provider needed
const { items, total, addItem } = useCart()
import { signal, spring } from 'what-framework'

function AnimatedBox() {
  const pos = spring(0, {
    stiffness: 170,
    damping: 26
  })

  return (
    <div>
      <div
        style={() => `transform: translateX(${pos()}px)`}
        class="box"
      />
      <button onclick={() => pos.set(200)}>
        Animate
      </button>
    </div>
  )
}
