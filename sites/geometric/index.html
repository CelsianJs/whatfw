import { h, mount, signal } from 'what-framework';

function Counter() {
  const count = signal(0);

  return h('button', {
    onClick: () => count.set(c => c + 1)
  }, 'Clicked ', () => count(), ' times');
}

mount(h(Counter), '#app');
