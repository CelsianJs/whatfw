Simple by design

Code that reads like vanilla

No magic. No special syntax. Just functions that return elements.

import { h, mount, signal } from 'what-framework';

function Counter() {
  const count = signal(0);

  return h('button', {
    onClick: () => count.set(c => c + 1)
  }, 'Clicked ', () => count(), ' times');
}

mount(h(Counter), '#app');
